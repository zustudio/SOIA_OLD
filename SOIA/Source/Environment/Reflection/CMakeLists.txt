cmake_minimum_required(VERSION 3.1)

subproject(Reflection)
	depends_on_projects(Environment Core_E Log CompileTime MathematicsCore)

in_sub(Element)
	add_PrivatePublic(
		ElementID
		ElementReflection
		ElementReflectionProvider
		RAbstractClass
		RClass
		RContainer
		TEST RElement
		RElementBase
		RPointer)
	add_Public(
		RWrapper
		RWrapperInterface
		RClassTemplate
		ReflectionMacros)
in_sub(Atom)
	add_PrivatePublic(
		Atom
		AtomReflection
		AtomReflectionProvider
		ReflectionFactory)
	add_Public(
		SimpleAtomReflection
		PointerReflectionTemplate
		VectorAtomReflectionTemplate)
in_sub(Function)
	add_PrivatePublic(
		RFunctionTemplate
		RFunction
		Function)
in_sub(Type)
	add_Public(
		DestructorFunctionTemplate
		TypeName
		MemberMirrorTemplate
		ObjectMirrorTemplate)
	add_PrivatePublic(
		TypeID
		PointerCounter
		TEST VoidPointer
		MemberMirror
		ObjectMirror)
in_sub(Unresolved)
	add_Public(
		ResolveInfoInterface)
	add_PrivatePublic(
		ResolvePointerInfo
		ResolveVectorInfo
		VoidPointerToUnresolvedRObject)
in_sub(Global)
	add_PrivatePublic(
		ContainerProvider
		ReflectionProviders)
end_sub()


auto_add_shared_library()
auto_add_unittests()
auto_static_check()
auto_include_directories()
auto_export_directories()
